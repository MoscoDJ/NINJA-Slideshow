            background: black;
        }

        .slide {
            position: absolute;
            top: 0;
            left: 0;
            width: 100%;
            height: 100%;
            opacity: 0;
            transition: opacity 1s;
            object-fit: contain;
        }
        .slide.active {
            opacity: 1;
        }
    </style>

<body>
    <div id="slideshow"></div>
    <script src="/socket.io/socket.io.js"></script>
    <script>
        const socket = io();
        let files = [];
        let currentIndex = 0;
        let timer = null;

        async function loadFiles() {
            const response = await fetch('/api/files');
            files = await response.json();
            if (files.length > 0) {
                showSlide(0);
            }
        }

        function showSlide(index) {
            const slideshow = document.getElementById('slideshow');
            slideshow.innerHTML = '';

            const file = files[index];
            const element = file.type === '.mp4' ? 'video' : 'img';
            const slide = document.createElement(element);
            
            slide.src = file.url;
            slide.className = 'slide active';
            
            if (element === 'video') {
                slide.autoplay = true;
                slide.onended = () => nextSlide();
            }
            
            slideshow.appendChild(slide);

            if (element === 'img') {
                clearTimeout(timer);
                timer = setTimeout(nextSlide, 15000);
            }
        }

        function nextSlide() {
            currentIndex = (currentIndex + 1) % files.length;
            showSlide(currentIndex);
        }

        socket.on('filesUpdated', loadFiles);
        loadFiles();
    </script>
</body>
</html>

<!-- public/admin.html -->
<!DOCTYPE html>
<html>
<head>
    <title>Slideshow Admin</title>
    <link href="https://cdnjs.cloudflare.com/ajax/libs/dragula/3.7.2/dragula.min.css" rel="stylesheet">
    <style>
        .file-list {
            display: grid;
            grid-template-columns: repeat(auto-fill, minmax(200px, 1fr));
            gap: 20px;
            padding: 20px;
        }
        .file-item {
            border: 1px solid #ccc;
            padding: 10px;
            cursor: move;
        }
        .thumbnail {
            width: 100%;
            height: 150px;
            object-fit: cover;
        }
        .delete-btn {
            margin-top: 10px;
            background: #ff4444;
            color: white;
            border: none;
            padding: 5px 10px;
            cursor: pointer;
        }
    </style>
</head>
<body>
    <div>
        <input type="file" id="fileInput" accept="image/*,video/*">
        <button onclick="uploadFile()">Upload</button>
    </div>
    <div id="fileList" class="file-list"></div>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/dragula/3.7.2/dragula.min.js"></script>
    <script src="/socket.io/socket.io.js"></script>
    <script>
        const socket = io();
        let drake;

        async function loadFiles() {
            const response = await fetch('/api/files');
            const files = await response.json();
            const fileList = document.getElementById('fileList');
            fileList.innerHTML = '';
            
            files.forEach(file => {
                const item = document.createElement('div');
                item.className = 'file-item';
                item.dataset.filename = file.name;
                
                const thumbnail = file.type === '.mp4' ? 'video' : 'img';
                item.innerHTML = `
                    <${thumbnail} src="${file.url}" class="thumbnail" ${thumbnail === 'video' ? 'muted' : ''}>
                    <p>${file.name}</p>
                    <button class="delete-btn" onclick="deleteFile('${file.name}')">Delete</button>
                `;
                
                fileList.appendChild(item);
            });

            if (drake) {
                drake.destroy();
            }
            
            drake = dragula([fileList], {
                moves: function (el, container, handle) {
                    return !handle.classList.contains('delete-btn');
                }
            });

            drake.on('drop', updateOrder);
        }

        async function uploadFile() {
            const fileInput = document.getElementById('fileInput');
            const file = fileInput.files[0];
            if (!file) return;

            const formData = new FormData();
            formData.append('file', file);

            try {
                await fetch('/api/upload', {
                    method: 'POST',
                    body: formData
                });
            } catch (error) {
                console.error('Upload error:', error);
            }
        }

        async function deleteFile(filename) {
            try {
                await fetch(`/api/files/${filename}`, {
                    method: 'DELETE'
                });
            } catch (error) {
                console.error('Delete error:', error);
            }
        }

        async function updateOrder() {
            const items = document.querySelectorAll('.file-item');
            const order = Array.from(items).map(item => item.dataset.filename);
            
            try {
                await fetch('/api/order', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ order })
                });
            } catch (error) {
                console.error('Order update error:', error);
            }
        }

        socket.on('filesUpdated', loadFiles);
        loadFiles();
    </script>
</body>
</html>
