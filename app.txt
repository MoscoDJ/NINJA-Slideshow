// server.js
const express = require('express');
const multer = require('multer');
const aws = require('aws-sdk');
const path = require('path');
const { Server } = require('socket.io');
const http = require('http');

const app = express();
const server = http.createServer(app);
const io = new Server(server);

// Configuración de Digital Ocean Spaces
const spacesEndpoint = new aws.Endpoint('your-region.digitaloceanspaces.com');
const s3 = new aws.S3({
  endpoint: spacesEndpoint,
  accessKeyId: process.env.SPACES_KEY,
  secretAccessKey: process.env.SPACES_SECRET
});

const BUCKET_NAME = 'your-bucket-name';
const FOLDER_NAME = 'slideshow';

// Configuración de multer para subida de archivos
const upload = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: 100 * 1024 * 1024 // 100MB limit
  }
});

// Rutas principales
app.use(express.static('public'));
app.use(express.json());

// Ruta para la página principal (slideshow)
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

// Ruta para el panel de administración
app.get('/admin', (req, res) => {
  res.sendFile(path.join(__dirname, 'public', 'admin.html'));
});

// Obtener lista de archivos
app.get('/api/files', async (req, res) => {
  try {
    const params = {
      Bucket: BUCKET_NAME,
      Prefix: FOLDER_NAME + '/'
    };
    
    const data = await s3.listObjects(params).promise();
    const files = data.Contents
      .filter(item => item.Size > 0) // Excluir la carpeta misma
      .map(item => ({
        name: path.basename(item.Key),
        url: `https://${BUCKET_NAME}.${spacesEndpoint.hostname}/${item.Key}`,
        type: path.extname(item.Key).toLowerCase()
      }));
    
    res.json(files);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Subir archivo
app.post('/api/upload', upload.single('file'), async (req, res) => {
  try {
    const file = req.file;
    const params = {
      Bucket: BUCKET_NAME,
      Key: `${FOLDER_NAME}/${file.originalname}`,
      Body: file.buffer,
      ACL: 'public-read',
      ContentType: file.mimetype
    };
    
    await s3.upload(params).promise();
    io.emit('filesUpdated');
    res.json({ message: 'File uploaded successfully' });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Eliminar archivo
app.delete('/api/files/:filename', async (req, res) => {
  try {
    const params = {
      Bucket: BUCKET_NAME,
      Key: `${FOLDER_NAME}/${req.params.filename}`
    };
    
    await s3.deleteObject(params).promise();
    io.emit('filesUpdated');
    res.json({ message: 'File deleted successfully' });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Actualizar orden de archivos
app.post('/api/order', async (req, res) => {
  try {
    // Aquí implementarías la lógica para guardar el orden
    // Podrías guardar un archivo JSON en el Space con el orden
    io.emit('filesUpdated');
    res.json({ message: 'Order updated successfully' });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

server.listen(3000, () => {
  console.log('Server running on port 3000');
});
